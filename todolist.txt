<script>
        // Step 3: Fetch the JSON data from the Flask route
        fetch('/your-endpoint')
            .then(response => response.json())  // Parse the JSON response
            .then(studyData => {
                // Now `studyData` is a JavaScript object with the data

                // Example usage of studyData in JavaScript
                console.log(studyData); // For debugging, to see the structure in the console

                // You can access different parts of studyData like so:
                studyData.classes_data.forEach((courseName, index) => {
                    console.log(`Course: ${courseName}, Total Hours: ${studyData.hours_data[index]}`);
                });

                // Example: Log weekly hours data for each course
                for (const [courseName, weeklyData] of Object.entries(studyData.study_hours_data)) {
                    console.log(`Course: ${courseName}`);
                    weeklyData.forEach(entry => {
                        console.log(`  Week ${parseInt(entry.week) + 1}: ${entry.weekly_hours} hours`);
                    });
                }
            })
            .catch(error => console.error('Error fetching data:', error));
    </script>
</body>
